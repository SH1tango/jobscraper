docker inspect addon_19263f11_jobwatcher | grep -A5 Mounts

docker run -it --rm --entrypoint sh 19263f11/amd64-addon-jobwatcher:0.1.xx

cat to open a file

list all containers
docker ps -a

docker ps -a | grep jobwatcher
